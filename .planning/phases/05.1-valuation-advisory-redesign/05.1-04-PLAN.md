    B. ADD `buildDynamicTable(container, initialRows)` helper INSIDE `_buildIntakeForms()` (placed below the `buildSimpleForm` helper definition around line 2316, so all DOM helpers are co-located). Implementation per UI-SPEC and PATTERNS.md:

       ```javascript
       function buildDynamicTable(container, initialRows) {
         // initialRows: array of {label, amount, rationale} from ebitda_adjustments
         var rowsContainer = mk('div', {'id': 'norm-rows'});
         var emptyMsg = mk('p',
           {'id': 'norm-empty',
            'style': 'font-size:.875rem;color:var(--muted);text-align:center;padding:1rem;display:none'},
           'No normalisation items. Click "+ Add normalisation item" to add one.');

         var rowCount = 0;
         function updateEmptyState() {
           emptyMsg.style.display = rowsContainer.children.length === 0 ? 'block' : 'none';
         }

         function addRow(label, amount, rationale) {
           var idx = rowCount++;
           var row = mk('div',
             {'class': 'norm-row', 'data-row-index': String(idx),
              'style': 'display:grid;grid-template-columns:2fr 120px 2fr auto;gap:.5rem;margin-bottom:.5rem;align-items:start'});

           var lInp = styledInput('norm_label_' + idx, 'text',
                                   {'placeholder': 'e.g. Owner salary above market'});
           lInp.value = label || '';
           var aInp = styledInput('norm_amount_' + idx, 'number',
                                   {'step': '1000', 'placeholder': '0'});
           aInp.value = (amount === 0 || amount) ? String(amount) : '';
           var rInp = styledInput('norm_rationale_' + idx, 'text',
                                   {'placeholder': 'Brief explanation'});
           rInp.value = rationale || '';

           var delBtn = mk('button',
             {'type': 'button', 'class': 'btn btn-sm btn-danger',
              'style': 'padding:.4rem .6rem;background:var(--red);color:white;border:none;border-radius:4px;cursor:pointer;font-size:.75rem'},
             'Remove Item');
           delBtn.addEventListener('click', function() {
             row.remove();
             updateEmptyState();
           });

           row.appendChild(lInp);
           row.appendChild(aInp);
           row.appendChild(rInp);
           row.appendChild(delBtn);
           rowsContainer.appendChild(row);
         }

         (initialRows || []).forEach(function(r) { addRow(r.label, r.amount, r.rationale); });

         var addBtn = mk('button',
           {'type': 'button', 'class': 'btn btn-secondary',
            'style': 'width:100%;margin-top:8px;padding:.5rem;font-size:.8rem;background:#f5f7fa;border:1px dashed var(--border);border-radius:4px;cursor:pointer'},
           '+ Add normalisation item');
         addBtn.addEventListener('click', function() {
           addRow('', '', '');
           updateEmptyState();
         });

         container.appendChild(rowsContainer);
         container.appendChild(emptyMsg);
         container.appendChild(addBtn);
         updateEmptyState();
       }
       ```
