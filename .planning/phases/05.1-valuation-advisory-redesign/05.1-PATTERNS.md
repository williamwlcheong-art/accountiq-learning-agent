**Research loop pattern** (from AI-SPEC Section 4, verified against SDK 0.96.0):
```python
def run_research_loop_sync(company_name, company_location, industry_sector, max_retries=2):
    client = anthropic.Anthropic(api_key=os.environ["ANTHROPIC_API_KEY"])
    messages = [{"role": "user", "content": user_prompt}]
    iteration = 0
    max_iterations = 5
    while iteration < max_iterations:
        response = client.messages.create(
            model="claude-sonnet-4-6",
            max_tokens=8000,
            system=RESEARCH_SYSTEM_PROMPT,
            tools=[WEB_SEARCH_TOOL],
            messages=messages,
        )
        iteration += 1
        if response.stop_reason == "end_turn":
            break
        if response.stop_reason == "pause_turn":
            messages.append({"role": "assistant", "content": response.content})
            messages.append({"role": "user", "content": "Please continue and return the JSON brief when ready."})
            continue
        if response.stop_reason == "max_tokens":
            raise RuntimeError("Research loop hit max_tokens")
        raise RuntimeError(f"Unexpected stop_reason: {response.stop_reason!r}")
    else:
        raise RuntimeError(f"Research loop exceeded {max_iterations} iterations without end_turn")
    brief = parse_and_validate_brief(response)
    return brief
```

**New function to add** — `compute_wacc_scenarios()` (from AI-SPEC Section 4b.1):
```python
def compute_wacc_scenarios(risk_free_rate: float, industry_beta: float, erp: float) -> dict:
    """
    Inputs are in PERCENT (e.g. risk_free_rate=4.65, erp=5.94, industry_beta=1.08).
    Returns WACC scenarios in PERCENT.
    Divide by 100 before passing to compute_dcf() — compute_dcf() takes decimal.
    """
    beta_spread = 0.15   # ±15% beta variation
    erp_spread = 0.25    # ±0.25 percentage point ERP variation
    mid = risk_free_rate + (industry_beta * erp)
    return {
        "high": round(mid + beta_spread * erp + erp_spread, 2),
        "mid":  round(mid, 2),
        "low":  round(mid - beta_spread * erp - erp_spread, 2),
    }
```

function buildDynamicTable(container, initialRows) {
  // initialRows: [{label, amount, rationale}] from ebitda_adjustments API fetch
  var tableWrap = mk('div', {'class': 'norm-table-wrap'});
  var rowsContainer = mk('div', {'id': 'norm-rows'});

  function addRow(label, amount, rationale, idx) {
    var row = mk('div', {'class': 'norm-row', 'data-row-index': String(idx),
                         'style': 'display:grid;grid-template-columns:2fr 1fr 2fr auto;gap:.5rem;margin-bottom:.5rem'});
    var lInp = styledInput('norm_label_' + idx, 'text', {});
    lInp.value = label || '';
    var aInp = styledInput('norm_amount_' + idx, 'number', {'step': '1'});
    aInp.value = amount || '';
    var rInp = styledInput('norm_rationale_' + idx, 'text', {});
    rInp.value = rationale || '';
    var delBtn = mk('button', {'type': 'button', 'style': 'padding:.3rem .6rem'}, '×');
    delBtn.addEventListener('click', function() { row.remove(); });
    row.appendChild(lInp); row.appendChild(aInp); row.appendChild(rInp); row.appendChild(delBtn);
    rowsContainer.appendChild(row);
  }

  var rowCount = initialRows.length;
  initialRows.forEach(function(r, i) { addRow(r.label, r.amount, r.rationale, i); });

  var addBtn = mk('button', {'type': 'button', 'class': 'btn btn-secondary',
                              'style': 'margin-top:.5rem;font-size:.8rem'}, '+ Add item');
  addBtn.addEventListener('click', function() { addRow('', '', '', rowCount++); });

  tableWrap.appendChild(rowsContainer);
  tableWrap.appendChild(addBtn);
  container.appendChild(tableWrap);
}
