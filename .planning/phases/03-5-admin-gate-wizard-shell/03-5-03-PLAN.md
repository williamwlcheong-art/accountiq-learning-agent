```python
# ---------------------------------------------------------------------------
# Wizard — authenticated non-admin upload path (Phase 3.5, D-05, D-06)
# ---------------------------------------------------------------------------

@app.post("/wizard/upload", status_code=201)
async def wizard_upload(
    background_tasks: BackgroundTasks,
    business_name: str = Form(...),
    file: UploadFile = File(...),
    db: aiosqlite.Connection = Depends(get_db),
    current_user: dict = Depends(get_current_user),   # NOT require_admin — per D-05
):
    """Create company + upload document for non-admin users. Reuses _run_ingestion."""
    name = business_name.strip()
    if not name:
        raise HTTPException(400, "Business name is required")

    suffix = Path(file.filename).suffix.lower()
    allowed = {".pdf", ".xlsx", ".xls", ".xlsm"}
    if suffix not in allowed:
        raise HTTPException(400, f"Only PDF and Excel files are accepted. Got: {suffix}")

    # Idempotent company creation — reuses existing helper (D-06)
    company_id, _ = await _resolve_or_create_company(db, name, current_user["id"])

    # Save file into company directory (project security rule: Path(file.filename).name)
    company_dir = PDF_DIR / str(company_id)
    company_dir.mkdir(exist_ok=True)
    safe_name = Path(file.filename).name
    dest = company_dir / safe_name
    with open(dest, "wb") as f:
        shutil.copyfileobj(file.file, f)

    # Insert document record
    async with db.execute("""
        INSERT INTO documents
            (company_id, filename, filepath, report_type, entity_type, fiscal_year_end, user_id)
        VALUES (?, ?, ?, 'compilation', 'sme', '', ?)
    """, (company_id, safe_name, str(dest), current_user["id"])) as cur:
        document_id = cur.lastrowid
    await db.commit()

    # Kick off background ingestion — same task as admin upload (D-06)
    background_tasks.add_task(
        _run_ingestion, document_id, company_id, str(dest), "sme", "Private", ""
    )

    return {"company_id": company_id, "document_id": document_id, "status": "processing"}
```

const WIZARD_REPORT_TYPES = [

function showWizard(user) {
  _wizardUser = user;
  document.getElementById('auth-page').style.display = 'none';
  document.getElementById('main-app').style.display = 'none';
  document.getElementById('wizard-page').style.display = 'block';
  // Set user email in wizard header — .textContent enforces XSS rule (AUTH-03)
  document.getElementById('wizard-user-email').textContent = user.email;
  renderWizardStep(1);
}

function renderWizardStep(step) {
  _wizardStep = step;
  document.querySelectorAll('.wizard-step').forEach(function(el) {
    el.style.display = (parseInt(el.dataset.step, 10) === step) ? 'block' : 'none';
  });
  const ind = document.getElementById('wizard-step-indicator');
  if (ind) ind.textContent = 'Step ' + step + ' of 3';

  // Step 2 setup: render report type cards
  if (step === 2) {
    _renderReportTypeCards();
  }
  // Step 3 setup: populate confirmation email
  if (step === 3 && _wizardUser) {
    // .textContent — XSS rule (AUTH-03)
    document.getElementById('wiz-confirm-email').textContent = _wizardUser.email;
  }
}

function _renderReportTypeCards() {
  const container = document.getElementById('wizard-report-cards');
  container.innerHTML = '';
  WIZARD_REPORT_TYPES.forEach(function(rt) {
    const card = document.createElement('div');
    card.style.cssText = 'border:2px solid var(--border);border-radius:8px;padding:.875rem 1rem;cursor:pointer;transition:border-color .15s,background .15s';
    card.dataset.key = rt.key;
    card.onclick = function() { _selectReportType(rt.key); };

    const nameEl = document.createElement('div');
    nameEl.style.cssText = 'font-weight:700;color:var(--navy);margin-bottom:.25rem';
    nameEl.textContent = rt.name;  // .textContent — XSS rule

    const descEl = document.createElement('div');
    descEl.style.cssText = 'font-size:.85rem;color:#555;line-height:1.4';
    descEl.textContent = rt.desc;  // .textContent — XSS rule

    card.appendChild(nameEl);
    card.appendChild(descEl);
    container.appendChild(card);
  });
  // Restore previously selected type if user navigated back
  if (_wizardReportType) {
    _selectReportType(_wizardReportType);
  }
}

function _selectReportType(key) {
  _wizardReportType = key;
  document.querySelectorAll('#wizard-report-cards > div').forEach(function(card) {
    const selected = card.dataset.key === key;
    card.style.borderColor = selected ? 'var(--blue)' : 'var(--border)';
    card.style.background = selected ? '#e3f2fd' : '';
  });
  const continueBtn = document.getElementById('wiz-step2-continue');
  if (continueBtn) continueBtn.disabled = false;
}

async function wizardSubmitStep1() {
  const nameInput = document.getElementById('wiz-business-name');
  const fileInput = document.getElementById('wiz-file');
  const name = nameInput ? nameInput.value.trim() : '';
  const file = fileInput && fileInput.files.length > 0 ? fileInput.files[0] : null;

  if (!name) {
    showAlert('wizard-alert', 'Business name is required.', 'error');
    return;
  }
  if (!file) {
    showAlert('wizard-alert', 'Please select a file to upload.', 'error');
    return;
  }

  const btn = event.currentTarget || document.querySelector('[onclick="wizardSubmitStep1()"]');
  const origText = btn ? btn.textContent : '';
  if (btn) { btn.disabled = true; btn.textContent = 'Uploading…'; }

  try {
    const fd = new FormData();
    fd.append('business_name', name);
    fd.append('file', file);
    const res = await fetch(API + '/wizard/upload', { method: 'POST', body: fd, credentials: 'include' });
    if (res.status === 401) { showAuthWall(true); return; }
    const json = await res.json();
    if (!res.ok) {
      showAlert('wizard-alert', json.detail || 'Upload failed. Please try again.', 'error');
      return;
    }
    _wizardUploadResult = json;
    showAlert('wizard-alert', '', 'success');  // clear any prior error
    renderWizardStep(2);
  } catch (e) {
    showAlert('wizard-alert', 'Connection error. Please check your network and try again.', 'error');
  } finally {
    if (btn) { btn.disabled = false; btn.textContent = origText || 'Continue →'; }
  }
}

function wizardConfirm() {
  if (!_wizardReportType) {
    showAlert('wizard-alert', 'Please select a report type to continue.', 'error');
    return;
  }
  renderWizardStep(3);
}

function wizardReset() {
  _wizardStep = 1;
  _wizardUploadResult = null;
  _wizardReportType = null;
  // Clear form inputs
  const nameInput = document.getElementById('wiz-business-name');
  const fileInput = document.getElementById('wiz-file');
  if (nameInput) nameInput.value = '';
  if (fileInput) fileInput.value = '';
  showAlert('wizard-alert', '', 'success');  // clear alerts
  renderWizardStep(1);
}

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| HTTP client → /wizard/upload | Unauthenticated or non-admin users submit business_name (free text) + file upload |
| Frontend JS → DOM | User email and business name may be rendered in wizard steps |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-5-08 | Tampering | Path traversal in /wizard/upload file save | mitigate | `Path(file.filename).name` strips directory components — established project security convention (AUTH-02); enforced in wizard_upload before file.open() |
| T-03-5-09 | Tampering | XSS via wizard dynamic values (email, report type name) in confirmation step | mitigate | All dynamic values use `.textContent` / `createTextNode()` in all wizard JS functions; `.innerHTML` is prohibited for user-controlled data (AUTH-03). Acceptance criteria includes grep for `\.innerHTML.*user` = 0 |
| T-03-5-10 | Elevation of Privilege | Non-admin user bypasses wizard and calls admin route directly | mitigate | Backend require_admin gate (from Plan 02) returns 403 regardless of frontend; frontend gating is UX-only, not security |
| T-03-5-11 | Spoofing | Wizard frontend relies on /auth/me is_admin for routing | accept | is_admin is authoritative in the backend (Plan 02); frontend branching is cosmetic; backend always enforces — ASVS V4 satisfied by backend gate not frontend check |
</threat_model>
