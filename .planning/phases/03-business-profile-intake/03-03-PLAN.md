// ===== Phase 3: Business profile panel =====

function toggleProfilePanel(companyId) {
  document.querySelectorAll('.profile-panel-row').forEach(p => {
    if (parseInt(p.dataset.companyId, 10) !== companyId) p.classList.remove('open');
  });
  const row = document.getElementById(`profile-panel-${companyId}`);
  if (!row) return;
  const isOpen = row.classList.toggle('open');
  if (isOpen) loadProfilePanel(companyId);
}

function _findCachedCompany(companyId) {
  const cache = window.__companiesCache || [];
  return cache.find(c => c.id === companyId) || null;
}

async function loadProfilePanel(companyId) {
  const td = document.getElementById(`profile-panel-td-${companyId}`);
  if (!td) return;
  // Loading placeholder
  td.textContent = '';
  const loading = document.createElement('div');
  loading.className = 'profile-panel-inner';
  loading.textContent = 'Loading…';
  td.appendChild(loading);

  const [members, adjustments, status] = await Promise.all([
    apiFetch(`/companies/${companyId}/management-team`),
    apiFetch(`/companies/${companyId}/ebitda-adjustments`),
    apiFetch(`/companies/${companyId}/profile-status`),
  ]);
  if (!members || !adjustments || !status) return;
  const company = _findCachedCompany(companyId) || {};

  td.textContent = '';
  const inner = document.createElement('div');
  inner.className = 'profile-panel-inner';

  // Panel header with Close button
  const header = document.createElement('div');
  header.className = 'flex';
  header.style.marginBottom = '1rem';
  const headerTitle = document.createElement('div');
  headerTitle.style.fontSize = '1rem';
  headerTitle.style.fontWeight = '600';
  headerTitle.style.color = 'var(--navy)';
  headerTitle.textContent = 'Business Profile';
  header.appendChild(headerTitle);
  const closeBtn = document.createElement('button');
  closeBtn.className = 'btn btn-sm ml-auto';
  closeBtn.style.background = '#eee';
  closeBtn.textContent = 'Close';
  closeBtn.onclick = () => toggleProfilePanel(companyId);
  header.appendChild(closeBtn);
  inner.appendChild(header);

  inner.appendChild(_renderIndustrySection(companyId, company));
  inner.appendChild(_renderDescriptionSection(companyId, company));
  inner.appendChild(_renderManagementTeamSection(companyId, members));
  inner.appendChild(_renderEbitdaSection(companyId, adjustments, status));

  td.appendChild(inner);
}

async function saveIndustry(companyId) {
  const sel = document.getElementById(`prof-industry-${companyId}`);
  const fd = new FormData();
  fd.append('sector', sel.value || '');
  const res = await apiPost(`/companies/${companyId}/profile`, fd);

function _formatAmount(v) {
  // Returns a span that mirrors numCell formatting (red parens for negative)
  const span = document.createElement('span');
  if (v == null) { span.textContent = '—'; return span; }
  if (v < 0) {
    span.style.color = 'var(--red)';
    span.textContent = `($${Math.abs(v).toLocaleString()})`;
  } else {
    span.textContent = `$${v.toLocaleString()}`;
  }
  return span;
}

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| User-typed text → DOM | All name/title/bio/label/rationale rendered via `.textContent` or `createTextNode` — never `.innerHTML` |
| Browser → Backend | FormData submitted via `apiPost`/raw `fetch` with `credentials: 'include'`; JWT cookie carries auth |
| `window.confirm()` strings | Built via JS string concatenation — confirm() is not HTML rendering, no XSS vector even with embedded user text |

|-----------|----------|-----------|-------------|-----------------|

NO `innerHTML` is used for any of m.name, m.title, m.bio, a.label, a.rationale. |
